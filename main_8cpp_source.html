68  // Some cool and nifty algorithm
69  // What is the delta in position

71  int const del_y = goal.y - start.y;
72 
73  // What direction to move in for each dimension
74  int const del_x_sign = std::copysign(1.0, del_x);
75  int const del_y_sign = std::copysign(1.0, del_y);
76 
77  // Push start onto the path

79  path.value().push_back(start);

81  auto const is_occupied = [&occupancy_map](auto const x,
82  auto const y) -> bool {
83  return occupancy_map.at(Position{x, y}) == 255;
84  };
85 
86  auto const repeat = [](size_t n, auto f) {
87  while (n--) f();
88  };

90  auto const move = [&is_occupied, &path](int const x_dir, int const y_dir) {
91  if (!path) {
92  return;
93  }
94  if (is_occupied(path.value().back().x + x_dir,
95  path.value().back().y + y_dir)) {
96  path = std::nullopt;
97  return;
98  }
99  path.value().push_back(
100  {path.value().back().x + x_dir, path.value().back().y + y_dir});
101  };
102 
103  // Fails if there is any obstacle in the way
104  // Move horizontally
105  repeat(std::abs(del_x), [&move, &del_x_sign]() { move(del_x_sign, 0); });
106  // Move vertically
107  repeat(std::abs(del_y), [&move, &del_y_sign]() { move(0, del_y_sign); });
108
