85 std::optional<Path> generate_global_path(
86  Position const& start, Position const& goal,
87  Map<unsigned char> const& occupancy_map) { // Calculation
88  // Pre-checks
89  if (occupancy_map.empty()) {
90  return std::nullopt;
91  }
92  // Get the dimensions of the map for bounds checking
93  auto const [dim_x, dim_y] = occupancy_map.shape();
94 
95  if (start.x > dim_x) {
96  return std::nullopt;
97  }
98  if (start.y > dim_y) {
99  return std::nullopt;
100  }
101  if (goal.x > dim_x) {
102  return std::nullopt;
103  }
104  if (goal.y > dim_y) {
105  return std::nullopt;
106  }
107  // What is the delta in position
108  int const del_x = goal.x - start.x;
109  int const del_y = goal.y - start.y;
110 
111  // What direction to move in for each dimension
112  int const del_x_sign = std::copysign(1.0, del_x);
113  int const del_y_sign = std::copysign(1.0, del_y);
114 
115  // Push start onto the path
116  std::optional<Path> path = Path{start};
117 
118  auto const is_occupied = [&occupancy_map](auto const x,
119  auto const y) -> bool {
120  return occupancy_map.at(Position{x, y}) == 255;
121  };
122 
123  auto const repeat = [](size_t n, auto f) {
124  while (n--) f();
125  };
126 
127  auto const move = [&is_occupied, &path](int const x_dir, int const y_dir) {
128  if (!path) {
129  return;
130  }
131  if (is_occupied(path.value().back().x + x_dir,
132  path.value().back().y + y_dir)) {
133  path = std::nullopt;
134  return;
135  }
136  path.value().push_back(
137  {path.value().back().x + x_dir, path.value().back().y + y_dir});
138  };
139 
140  // Fails if there is any obstacle in the way
141  // Move horizontally
142  repeat(std::abs(del_x), [&move, &del_x_sign]() { move(del_x_sign, 0); });
143  // Move vertically
144  repeat(std::abs(del_y), [&move, &del_y_sign]() { move(0, del_y_sign); });
145 
146  return path;
147 }

158 std::optional<Map<unsigned char>> parseSetMapRequest(
159  std::shared_ptr<SetMap::Request> const request) {
160  if (request->map.layout.dim.empty() || request->map.data.empty()) {
161  return std::nullopt;

163  // Check that map layout makes sense
164  if ((request->map.layout.dim[0].size * request->map.layout.dim[1].size) !=
165  request->map.layout.dim[0].stride) {
166  return std::nullopt;

168  if (request->map.layout.dim[0].stride != request->map.data.size()) {
169  return std::nullopt;

172  auto const occupancy_map = request->map;
173  auto const begin = std::begin(occupancy_map.data);
174 
175  // Populate the map
176  Map<unsigned char> map;
177  for (size_t row = 0; row < occupancy_map.layout.dim[0].size; row++) {
178  auto const row_beginning = begin + row * occupancy_map.layout.dim[1].size;
179  auto const row_end = begin + (row + 1) * occupancy_map.layout.dim[1].size;
180  map.get_data().push_back({row_beginning, row_end});
181  }
