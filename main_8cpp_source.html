71  Position const& start, Position const& goal,
72  Map<unsigned char> const& occupancy_map) { // Calculation
73  // Some cool and nifty algorithm
74  // What is the delta in position
75  int const del_x = goal.x - start.x;
76  int const del_y = goal.y - start.y;
77 
78  // What direction to move in for each dimension
79  int const del_x_sign = std::copysign(1.0, del_x);
80  int const del_y_sign = std::copysign(1.0, del_y);
81 
82  // Push start onto the path
83  std::optional<Path> path = Path{};
84  path.value().push_back(start);
85 
86  auto const is_occupied = [&occupancy_map](auto const x,
87  auto const y) -> bool {
88  return occupancy_map.at(Position{x, y}) == 255;
89  };
90 
91  auto const repeat = [](size_t n, auto f) {
92  while (n--) f();
93  };
94 
95  auto const move = [&is_occupied, &path](int const x_dir, int const y_dir) {
96  if (!path) {
97  return;
98  }
99  if (is_occupied(path.value().back().x + x_dir,
100  path.value().back().y + y_dir)) {
101  path = std::nullopt;
102  return;
103  }
104  path.value().push_back(
105  {path.value().back().x + x_dir, path.value().back().y + y_dir});
106  };
107 
108  // Fails if there is any obstacle in the way
109  // Move horizontally
110  repeat(std::abs(del_x), [&move, &del_x_sign]() { move(del_x_sign, 0); });
111  // Move vertically
112  repeat(std::abs(del_y), [&move, &del_y_sign]() { move(0, del_y_sign); });
113 
114  return path;
115 }
