46  // What is the delta in position
47  int const del_x = goal.x - start.x;
48  int const del_y = goal.y - start.y;
49 
50  // What direction to move in for each dimension
51  int const del_x_sign = std::copysign(1.0, del_x);
52  int const del_y_sign = std::copysign(1.0, del_y);
53 
54  // Push start onto the path
55  Path path;
56  path.push_back(start);
57 
58  auto is_occupied = [&costmap](auto const x, auto const y) -> bool {
59  return costmap.at(Position{x, y}) == 255;
60  };
61 
62  // Fails if there is any obstacle in the way
63  // Move horizontally
64  for (size_t i = 0; i < (std::abs(del_x)); ++i) {
65  if (is_occupied(path.back().x + del_x_sign, path.back().y)) {
66  return std::nullopt;
67  }
68  path.push_back({path.back().x + del_x_sign, path.back().y});
69  }
70  // Move vertically
71  for (size_t i = 0; i < (std::abs(del_y)); i++) {
72  if (is_occupied(path.back().x, path.back().y + del_y_sign)) {
73  return std::nullopt;
74  }
75  path.push_back({path.back().x, path.back().y + del_y_sign});
76  }
