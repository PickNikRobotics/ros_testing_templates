217  // From the start and goal, generate a trajectory (Deterministic calculation)
218  Path generate_global_path(Position const& start,
219  Position const& goal) { // Calculation
220  // Some cool and nifty algorithm
221  // What is the delta in position
222  int const del_x = goal.x - start.x;
223  int const del_y = goal.y - start.y;
224 
225  // What direction to move in for each dimension
226  int const del_x_sign = std::copysign(1.0, del_x);
227  int const del_y_sign = std::copysign(1.0, del_y);
228 
229  // Push start onto the path
230  Path path;
231  path.push_back(start);
232 
233  // Fails if there is any obstacle in the way
234  // Move horizontally
235  // x limit is reduced by half of the robot size to prevent the
236  // algorithm from checking collision outside of the map
237  auto const x_limit = std::abs(del_x) - std::floor(robot_size_ / 2);
238  for (size_t i = 0; i < x_limit; ++i) {
239  if ((*is_occupied_)(map_, path.back().x + del_x_sign, path.back().y)) {
240  std::cout << "Failed horizontally at " << path.back().x + del_x_sign
241  << " " << path.back().y << "\n";
242  return {};
243  }
244  path.push_back({path.back().x + del_x_sign, path.back().y});
245  }
246  // Move vertically
247  auto const y_limit = std::abs(del_y) - std::floor(robot_size_ / 2);
248  for (size_t i = 0; i < y_limit; i++) {
249  if ((*is_occupied_)(map_, path.back().x, path.back().y + del_y_sign)) {
250  std::cout << "Failed vertically at " << path.back().x << " "
251  << path.back().y + del_y_sign << "\n";
252  return {};
253  }
254  path.push_back({path.back().x, path.back().y + del_y_sign});
255  }
256 
257  return path;
258  }
