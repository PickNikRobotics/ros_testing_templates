162  // From the start and goal, generate a trajectory (Deterministic calculation)
163  std::optional<Path> generate_global_path(Position const& start, Position const& goal) { // Calculation
164  // Some cool and nifty algorithm
165  // What is the delta in position
166  int const del_x = goal.x - start.x;
167  int const del_y = goal.y - start.y;
168 
169  // What direction to move in for each dimension
170  int const del_x_sign = std::copysign(1.0,del_x);
171  int const del_y_sign = std::copysign(1.0,del_y);
172 
173  // Push start onto the path
174  Path path;
175  path.push_back(start);
176 
177  auto is_occupied = [this](auto const x, auto const y) -> bool {
178  return this->map_.at(Position{x, y}) == 1;
179  };
180 
181  // Fails if there is any obstacle in the way
182  // Move horizontally
183  for (size_t i = 0; i < (std::abs(del_x)); ++i) {
184  if (is_occupied(path.back().x + del_x_sign, path.back().y)) {
185  return std::nullopt;
186  }
187  path.push_back({path.back().x + del_x_sign, path.back().y});
188  }
189  // Move vertically
190  for (size_t i = 0; i < (std::abs(del_y)); i++) {
191  // if (costmap.at(Position{path.back().x, path.back().y + del_y_sign}) == 1) {
192  if (is_occupied(path.back().x, path.back().y + del_y_sign)) {
193  return std::nullopt;
194  }
195  path.push_back({path.back().x, path.back().y + del_y_sign});
196  }
197 
198  return path;
199  }
