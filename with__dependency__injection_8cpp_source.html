216  std::optional<Path> generate_global_path(
217  Position const& start, Position const& goal) { // Calculation
218  // Some cool and nifty algorithm
219  // What is the delta in position
220  int const del_x = goal.x - start.x;
221  int const del_y = goal.y - start.y;
222 
223  // What direction to move in for each dimension
224  int const del_x_sign = std::copysign(1.0, del_x);
225  int const del_y_sign = std::copysign(1.0, del_y);
226 
227  // Push start onto the path
228  Path path;
229  path.push_back(start);
230 
231  auto is_occupied = [this](auto const x, auto const y) -> bool {
232  return this->map_.at(Position{x, y}) == 1;
233  };
234 
235  // Fails if there is any obstacle in the way
236  // Move horizontally
237  for (size_t i = 0; i < (std::abs(del_x)); ++i) {
238  if (is_occupied(path.back().x + del_x_sign, path.back().y)) {
239  return std::nullopt;
240  }
241  path.push_back({path.back().x + del_x_sign, path.back().y});
242  }
243  // Move vertically
244  for (size_t i = 0; i < (std::abs(del_y)); i++) {
245  // if (costmap.at(Position{path.back().x, path.back().y + del_y_sign})
246  // == 1) {
247  if (is_occupied(path.back().x, path.back().y + del_y_sign)) {
248  return std::nullopt;
249  }
250  path.push_back({path.back().x, path.back().y + del_y_sign});
251  }
252 
253  return path;
